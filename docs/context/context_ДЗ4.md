# Контекст решения
<!-- Окружение системы (роли, участники, внешние системы) и связи системы с ним. Диаграмма контекста C4 и текстовое описание. 
Подробнее: https://confluence.mts.ru/pages/viewpage.action?pageId=375783261
-->
```plantuml
@startuml
!include https://raw.githubusercontent.com/plantuml-stdlib/C4-PlantUML/master/C4_Container.puml

LAYOUT_WITH_LEGEND()
System_Ext(es, "Система рассылки уведомлений", "Сервис Mail.ru") 

System_Boundary(c, "Авторизация пользователя"){
Person(pbc, "Пользователь", "Лицо желающее принять участие в конференции")

System(ibs, "Система авторизации пользователя", "Позволяет Пользователю зарегистрироваться для участия в конференции")


Rel(pbc, ibs, "Доступ к интерфейсу авторизации")
Rel(ibs, es, "Отправка e-mail пользоватю о регистрации")
}

System_Boundary(d, "Загрузка доклада"){
Person(Sp, "Докладчик", "Выступающий участник")
System(isp, "Загрузка презентации", "Позволяет Докладчику загрузить презентацию")

Rel(Sp, isp, "Доступ к интерфейсу загрузки презентаций")
Rel(isp, es, "Отправка e-mail Докдадчику", " о загрузке презентации")

}

System_Boundary(a, "Рецензирование доклада "){
Person(ap, "Администратор", "Участник не принимающий непосредственного участия в конференци, обладающий правами доступа ко всем данным системы")
System(asp, "Рецензирование доклада", "Администратор рецензирует доклад")

Rel(ap, asp, "Доступ к интерфейсу рецензирования презентаций")
Rel(asp, es, "Отправка e-mails Докдадчику о результате рецензирования")
}

System_Boundary(ass, "Формирование расписания"){
Person(as, "Администратор", "Участник не принимающий непосредственного участия в конференци, обладающий правами доступа ко всем данным системы")
System(csh, "Формирование расписания", "Для отрецинзированных докладов создается расписание конференции")

Rel(as, csh, "Доступ к интерфейсу формирования расписания")
Rel(csh, es, "Отправка e-mails Докдадчику о дате и времени выступдения")
}
Rel(ibs, isp, "Участник зарегистрирован как Докладчик")

Rel(isp, asp, "Доклад загружен и отправлен на рецензирование")

Rel(asp, csh, "Доклады отрецензированы")

@enduml
```
